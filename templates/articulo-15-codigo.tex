% articulo-15-codigo.tex - Código fuente: entornos y tokens de resaltado sintáctico.
%
% Pandoc genera estos entornos para bloques de código con resaltado:
%   \begin{Shaded}…\end{Shaded}       bloque con fondo gris
%   \begin{Highlighting}…             tokens de color dentro de Shaded
%
% El entorno Shaded envuelve Highlighting en shaded* (framed) con tamaño
% \small. El color de fondo (shadecolor) es gris muy claro RGB(248,248,248).
%
% Los tokens \XxxTok{} corresponden a las categorías de resaltado de Pandoc/KDE.
% No eliminar ninguno aunque no se use actualmente: pandoc puede emitirlos
% dependiendo del lenguaje y la versión del resaltador.

\usepackage{fancyvrb}
\usepackage{framed}

% shadecolor: gris neutro frío.
% \topsep a 0pt: elimina el espacio extra que framed añade antes y después del bloque.
% \setstretch{1}: interlineado fijo independiente del valor global del documento.
\definecolor{shadecolor}{RGB}{235,237,240}
\newenvironment{Shaded}{%
  \setlength{\topsep}{12pt}%
  \setstretch{1}%
  \small
  \begin{shaded*}%
}{\end{shaded*}}
\DefineVerbatimEnvironment{Highlighting}{Verbatim}{
  commandchars = \\\{\},
  fontsize     = \small
}
