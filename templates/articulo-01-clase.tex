% articulo-01-clase.tex - Clase base del documento.
%
% Variables YAML (todas opcionales; los valores indicados son los defectos):
%   fontsize:    12pt          Tamaño de cuerpo; cualquier valor válido para article.
%   papersize:   a4paper       Tamaño de papel; nombres de geometry: a4paper,
%                              letterpaper, a5paper, b5paper…
%   classoption: [twoside]     Opciones de clase como lista YAML. twoside produce
%                              márgenes alternos y encabezados distintos en recto
%                              y verso. oneside unifica ambos márgenes.
%                              Ejemplo con varias opciones:
%                                classoption: [oneside, draft]

\documentclass[%
  $if(fontsize)$
    $fontsize$
  $else$
    12pt
  $endif$,%
  $if(papersize)$
    $papersize$
  $else$
    a4paper
  $endif$,%
  $if(classoption)$
    $for(classoption)$
      $classoption$$sep$,
    $endfor$
  $else$
    twoside
  $endif$%
]{$if(documentclass)$$documentclass$$else$article$endif$}

$if(beamerarticle)$
\usepackage{beamerarticle}
$endif$
