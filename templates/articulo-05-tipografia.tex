% articulo-05-tipografia.tex - Tipografía fina, interlineado, párrafo y control de viudas.
%
% \ParIndent centraliza la sangría de primera línea (media pulgada, Chicago § 2.15).
% Se usa en: párrafos del cuerpo, notas al pie, listas (nivel 1), índice (nivel 3).
% Para cambiar la sangría en todo el documento, modificar solo este valor.
%
% Variable YAML:
%   linestretch: 1.5   Interlineado; 2.0 para doble espacio.

% ── Tipografía fina ───────────────────────────────────────────────────────────

\usepackage[protrusion=true, expansion=true, tracking=true]{microtype}

% ── Interlineado ──────────────────────────────────────────────────────────────

\usepackage{setspace}
\setstretch{$if(linestretch)$$linestretch$$else$1.5$endif$}

% ── Párrafo ───────────────────────────────────────────────────────────────────

\newlength{\ParIndent}
\setlength{\ParIndent}{1.27cm}
\setlength{\parindent}{\ParIndent}
\setlength{\parskip}{0pt}

% NO usar indentfirst: el primer párrafo tras título de sección no lleva sangría
% (norma Chicago § 2.15; en español es igualmente aceptable).
% \usepackage{indentfirst} desactivado intencionadamente

% ── Control de viudas y huérfanas ───────────────────────────────────────────────────────────
% Valor 10000 = penalización máxima (equivalente a prohibirlo).

\clubpenalty=10000
\widowpenalty=10000
\displaywidowpenalty=10000
