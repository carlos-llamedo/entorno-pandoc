\newcommand{\toc@line}[5]{%
  \begingroup
    \toc@font
    \leftskip=#1\relax
    \parindent=\z@
    \rightskip=\z@
    \parfillskip=\z@ plus 1fil
    \leavevmode
    \if\relax\detokenize{#3}\relax\else
      \llap{#2{#3}\kern1em}%
    \fi
    #2{#4}%
    \nobreak{\toc@font\enspace\textperiodcentered\enspace#5}%
    \par
  \endgroup
}

% Capítulo: versalitas.
\renewcommand*{\l@chapter}[2]{%
  \ifnum\c@tocdepth>\m@ne
    \addpenalty{-\@highpenalty}%
    \addvspace{0.4\baselineskip}%
    \begingroup
      \def\numberline##1{}%
      \toc@line{0pt}{\MakeUppercase}{}{#1}{#2}%
    \endgroup
  \fi}

% Sección: redonda.
\renewcommand*{\l@section}[2]{%
  \ifnum\c@tocdepth>\z@
    \begingroup
      \def\numberline##1{}%
      \toc@line{\tocsubindent}{\upshape}{}{#1}{#2}%
    \endgroup
  \fi}

% Subsección: cursiva.
\renewcommand*{\l@subsection}[2]{%
  \ifnum\c@tocdepth>\@ne
    \begingroup
      \def\numberline##1{}%
      \toc@line{2\tocsubindent}{\itshape}{}{#1}{#2}%
    \endgroup
  \fi}

% Subsubsección: cursiva, más sangrada.
\renewcommand*{\l@subsubsection}[2]{%
  \ifnum\c@tocdepth>\tw@
    \begingroup
      \def\numberline##1{}%
      \toc@line{3\tocsubindent}{\itshape}{}{#1}{#2}%
    \endgroup
  \fi}
